```python
class Solution:
    def findMaxConsecutiveOnes(self, nums: List[int]) -> int:
        a=[]
        count=0
        for i in range(len(nums)):
            if nums[i]==1:
                count+=1 
            else:
                a.append(count)
                count=0 
            if i==len(nums)-1:
                a.append(count)
        return max(a)
```
