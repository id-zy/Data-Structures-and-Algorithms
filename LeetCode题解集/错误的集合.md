class Solution:
    def findErrorNums(self, nums: List[int]) -> List[int]:
        n=len(nums)
        x,y=0,0
        nums_count=[0]*(n+1)
        for i in nums :
            nums_count[i]+=1
        for i in range(1,n+1):
            if nums_count[i]==2:
                x=i
            if nums_count[i]==0:
                y=i               
        return [x,y]
