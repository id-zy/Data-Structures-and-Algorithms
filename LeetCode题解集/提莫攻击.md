```python
class Solution:
    def findPoisonedDuration(self, timeSeries: List[int], duration: int) -> int:
        m=timeSeries
        n=duration
        l=len(m)
        count=0
        i=0
        while i<l-1:
            if m[i]+n-1<m[i+1]:
                count+=n
            else:
                count+=m[i+1]-m[i]
            i+=1
        return count+n
```
