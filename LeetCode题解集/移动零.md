```python
class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        l=len(nums)
        for i in range(l-1):
            d=i+1
            while nums[i]==0 and d<l:
                nums[i],nums[d]=nums[d],nums[i]
                d+=1
```
