```python
class Solution:
    def maximumSwap(self, num: int) -> int:
        digits = list(str(num))
        n = len(digits)
        last = {int(d): i for i, d in enumerate(digits)}
        for i in range(n):
            for d in range(9, int(digits[i]), -1):
                if last.get(d, -1) > i:  
                    j = last[d]
                    digits[i], digits[j] = digits[j], digits[i]
                    return int(''.join(digits))
        return num
```
