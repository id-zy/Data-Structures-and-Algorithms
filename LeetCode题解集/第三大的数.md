class Solution:
    def thirdMax(self, nums: List[int]) -> int:
        max2=max(nums)
        max1=max(nums)
        if len(nums)<3:
            return max1
        pid=0
        while 1 :
            nums.remove(max1)
            if pid ==2 or len(nums)==0:
                break
            if max1!=max(nums):
                pid+=1
                max1=max(nums)
        if pid<2:
            return max2
        else:
            return max1
