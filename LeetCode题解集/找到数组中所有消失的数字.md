```python
class Solution:
    def findDisappearedNumbers(self, nums: List[int]) -> List[int]:
        mp=dict()
        l=len(nums)
        a=[]
        for i in range(1,l+1):
             mp[i]=0
        for i in nums:
            mp[i]=i
        for i in range(1,l+1):
            if mp[i]==0:
                a.append(i)
        return a
```
