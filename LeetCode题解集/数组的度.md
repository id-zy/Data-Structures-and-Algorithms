```python
class Solution:
    def findShortestSubArray(self, nums: List[int]) -> int:
        info=dict()
        for i,num in enumerate(nums):
            if num in info:
                info[num][0]+=1
                info[num][2]=i
            else:
                info[num]=[1,i,i]
        du=minlen=0
        for count,left,right in info.values():
            if du < count:
                du=count
                minlen=right-left+1
            elif du==count:
                x=right-left+1
                if x< minlen:
                    minlen=x
        return minlen
```
