```python
class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        n=numRows
        L = [] 
        for i in range(n):
            line=[]
            for j in range(i+1):
                if j==0 or i==j:
                    line.append(1)
                else :
                    a=L[i-1][j-1]+L[i-1][j]
                    line.append(a)
            L.append(line)
        return L
```
