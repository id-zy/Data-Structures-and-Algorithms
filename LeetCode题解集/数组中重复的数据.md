class Solution:
    def findDuplicates(self, nums: List[int]) -> List[int]:
        ans=[]
        l=len(nums)
        for i in nums:
            idx=abs(i)-1
            if nums[idx]>0:
                nums[idx]=-nums[idx]
            else:
                ans.append(abs(i))
        return ans
